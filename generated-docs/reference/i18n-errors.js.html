            '404Lead': 'The page you requested is unavailable or has been moved.',
            '404Support': 'If you reached this page from a link on this site, please report the problem.'
        }
    };

    function resolveLocale(language) {
        if (typeof language !== 'string') {
            return 'en';
        }

        return language.toLowerCase().startsWith('uk') ? 'uk' : 'en';
    }

    runtimeWindow.LandingPageErrorMessages = {
        resolveLocale,
        getMessages(language) {
            const locale = resolveLocale(language);
            return messageCatalog[locale];
        }
    };
}(window));
